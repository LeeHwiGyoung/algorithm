#include<iostream>
using namespace std;
class frame {
private:
	int height, weight;
	int rank;
public:
	frame() {
		this->height = 0;
		this->weight = 0;
		this->rank = 1;
	}
	void setframe(int height, int weight) {
		this->height = height;
		this->weight = weight;
	}
	void compare(frame &a)
	{
		if (a.height < this->height && a.weight < this->weight)
			a.rank += 1;
		else if (a.height > this->height && a.weight > this->weight)
			this->rank += 1;
	}
	void print()
	{
		cout << this->rank << " ";
	}
};
int main()
{
	int num;
	int height, weight;	cin >> num;
	frame* arr = new frame[num];
	for (int i = 0; i < num; i++) {
		cin >> weight >> height;
		arr[i].setframe(height, weight);
	}
	for (int i = 0; i < num-1; i++)
	{
		for (int j = i+1; j < num; j++)
		{
			arr[i].compare(arr[j]);
		}
	}
	for (int i = 0; i < num; i++)
	{
		arr[i].print();
	}
}
